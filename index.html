Pikadaytime

Lightweight date and time picker based on Pikaday

Download this project as a .zip file Download this project as a tar.gz file

Examples

Simple date picker

Using default options. No initial date.

var picker = new Pikaday({field : document.getElementById("picker")});

12 hours format date and time picker

12 hours format with initial date. Position:right

var picker = new Pikaday({
    field          : document.getElementById("picker"),
    defaultDate    : new Date(),
    setDefaultDate : true,
    showTime       : true,
    hours24format   : false,
    position       : 'top right'
});

Date and time picker with one select for hours and minutes

Split hours and minutes pickers. Minutes step is 10.

var picker = new Pikaday({
    field          : document.getElementById("picker"),
    defaultDate    : new Date(),
    setDefaultDate : true,
    showTime       : true,
    splitTimeView  : true,
    minutesStep    : 10
});

Events handlers

See output in console

var picker = new Pikaday({
    field          : document.getElementById("picker"),
    defaultDate    : new Date(),
    setDefaultDate : true,
    showTime       : true,
    onSelect : function(date) {
        console.log('onSelect', date, this);
    },
    onOpen : function() {
        console.log('onOpen', this);
    },
    onClose : function() {
        console.log('onClose', this);
    },
    onDraw : function() {
        console.log('onDraw', this);
    }

});

Datepicker options and methods

See on Pikaday github page

Timepicker options

showTime : [Boolean] show time picker. (default false) hours24format : [Boolean] use 24 or 12 hours format. (default true (24 format)) showSeconds : [Boolean] show seconds picker. (default false) minutesStep : [Number] Show every n minutes. (default 1) secondsStep : [Number] Show every n seconds. (default 1) splitTimeView : [Number] Show one time selector with splited hours and minutes. Only 24 hours format. (default false)

Internationalization

You can setup i18n globaly.

Pikaday.prototype.i18n = {
    previousMonth : 'Предыдущий месяц',
    nextMonth     : 'Следующий месяц',
    months        : ['Январь','Февраль','Март','Апрель','Май','Июнь','Июль','Август','Сентябрь','Октябрь','Ноябрь','Декабрь'],
    weekdays      : ['Воскресенье', 'Понедельник','Вторник','Среда','Четверг','Пятница','Суббота'],
    weekdaysShort : ['Вс','Пн','Вт','Ср','Чт','Пт','Сб']
}

Or pass i18n in picker constructor

var picker = new Pikaday({
    field : document.getElementById('picker'),
    .....
    i18n : {
        previousMonth : '先月',
        nextMonth     : '来月',
        months        : ['１月','２月','３月','４月','５月','６月','７月','８月','９月','１０月','１１月','１２月'],
        weekdays      : ['日曜日', '月曜日','火曜日',　'水曜日', '木曜日','金曜日','土曜日'],
        weekdaysShort : ['日', '月','火',　'水', '木','金','土']
    }
});
