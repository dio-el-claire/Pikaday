Pikadaytime

Lightweight date and time picker based on http://dbushell.github.io/Pikaday

Download this project as a .zip file Download this project as a tar.gz file

Examples

Simple date picker

Using default options. No initial date.

var picker = new Pikaday({field : document.getElementById("picker")});

Date and time picker

12 hours format with initial date. Position:right

var picker = new Pikaday({
    field          : document.getElementById("picker"),
    defaultDate    : new Date(),
    setDefaultDate : true,
    showTime       : true,
    hours24format   : false,
    position       : 'top right'
});

Date and time picker

Split hours and minutes pickers. Minutes step is 10.

var picker = new Pikaday({
    field          : document.getElementById("picker"),
    defaultDate    : new Date(),
    setDefaultDate : true,
    showTime       : true,
    splitTimeView  : true,
    minutesStep    : 10
});

Events handlers

See output in console

var picker = new Pikaday({
    field          : document.getElementById("picker"),
    defaultDate    : new Date(),
    setDefaultDate : true,
    showTime       : true,
    onSelect : function(date) {
        console.log('onSelect', date, this);
    },
    onOpen : function() {
        console.log('onOpen', this);
    },
    onClose : function() {
        console.log('onClose', this);
    },
    onDraw : function() {
        console.log('onDraw', this);
    }

});
